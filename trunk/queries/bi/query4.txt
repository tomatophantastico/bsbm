  prefix bsbm: <http://www4.wiwiss.fu-berlin.de/bizer/bsbm/v01/vocabulary/>
  prefix bsbm-inst: <http://www4.wiwiss.fu-berlin.de/bizer/bsbm/v01/instances/>
  prefix xsd: <http://www.w3.org/2001/XMLSchema#>

  Select ?feature (?withFeaturePrice/?withoutFeaturePrice As ?priceRatio)
  {
    { Select ?feature (avg(xsd:float(?price)) As ?withFeaturePrice)
      {
        ?product a %ProductType% .
        ?product bsbm:productFeature ?feature .
        ?offer bsbm:product ?product .
        ?offer bsbm:price ?price .
      }
      Group By ?feature
    }
    { Select ?feature (avg(xsd:float(?price)) As ?withoutFeaturePrice)
      {
        ?product a %ProductType% .
        ?offer bsbm:product ?product .
        ?offer bsbm:price ?price .
        minus { ?product bsbm:productFeature ?feature }
      }
      Group By ?feature
    }
  }
  Order By desc(?withFeaturePrice/?withoutFeaturePrice)
  Limit 10
